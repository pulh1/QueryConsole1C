Функция ПолучитьДанные(ДатаНачала, ДатаОкончания)
	Запрос = Новый Запрос;
	Запрос.МенеджерВременныхТаблиц = Новый МенеджерВременныхТаблиц();
	Запрос.УстановитьПараметр("ДатаНачала", ДатаНачала);
	Запрос.УстановитьПараметр("ДатаОкончания", ДатаОкончания);
	Запрос.Текст = 
	"ВЫБРАТЬ ПЕРВЫЕ 10
	|	Сотрудники.Ссылка КАК Сотрудник,
	|	&ДатаНачала КАК ДатаНачала,
	|	&ДатаОкончания КАК ДатаОкончания,
	|	&ДатаОкончания КАК ДатаАктуальности
	|ПОМЕСТИТЬ ВТСотрудники
	|ИЗ
	|	Справочник.Сотрудники КАК Сотрудники
	|
	|УПОРЯДОЧИТЬ ПО
	|	Сотрудник";
	Запрос.Выполнить();
	ПараметрыПолученияДанных = УчетРабочегоВремениРасширенный.ПараметрыДляСоздатьВТПлановоеВремяСотрудников();
	
	ЗапросВспомогательный = Новый Запрос;
	ЗапросВспомогательный.МенеджерВременныхТаблиц = Запрос.МенеджерВременныхТаблиц;
	ЗапросВспомогательный.Текст = 
	"ВЫБРАТЬ РАЗЛИЧНЫЕ
	|	ВТФильтр.ДатаНачала,
	|	ВТФильтр.ДатаОкончания
	|ИЗ
	|	ВТФильтр КАК ВТФильтр";
	
	ЗапросВспомогательный.Текст = СтрЗаменить(ЗапросВспомогательный.Текст, "ВТФильтр", "ВТСотрудники");
	
	Выборка = ЗапросВспомогательный.Выполнить().Выбрать();
	ЕстьПериодыБольшеМесяца = Ложь;
	МинимальнаяДата = '39991231';
	МаксимальнаяДата = '00010101';
	Пока Выборка.Следующий() Цикл
		Если Выборка.ДатаНачала < МинимальнаяДата Тогда
			МинимальнаяДата = Выборка.ДатаНачала;
		КонецЕсли;
		Если Выборка.ДатаОкончания > МаксимальнаяДата Тогда
			МаксимальнаяДата = Выборка.ДатаОкончания;
		КонецЕсли;
		Если НачалоМесяца(Выборка.ДатаНачала) <> НачалоМесяца(Выборка.ДатаОкончания) Тогда
			ЕстьПериодыБольшеМесяца = Истина;
		КонецЕсли;
	КонецЦикла;
	
	Если ЕстьПериодыБольшеМесяца Тогда
		ЗарплатаКадрыОбщиеНаборыДанных.СоздатьВТПериоды(
			Запрос.МенеджерВременныхТаблиц, 
			МинимальнаяДата, 
			МаксимальнаяДата,
			"МЕСЯЦ",,
			"ВТПериодыДляВТФильтрИсполняемоеПредставлениеПлановоеВремя");	
	
		ТекстЗапросаВТПериоды = 
		"ВЫБРАТЬ
		|	Фильтр.Сотрудник КАК Сотрудник,
		|	ВЫБОР
		|		КОГДА НАЧАЛОПЕРИОДА(Периоды.Период, МЕСЯЦ) > Фильтр.ДатаНачала
		|			ТОГДА НАЧАЛОПЕРИОДА(Периоды.Период, МЕСЯЦ)
		|		ИНАЧЕ Фильтр.ДатаНачала
		|	КОНЕЦ КАК ДатаНачала,
		|	ВЫБОР
		|		КОГДА КОНЕЦПЕРИОДА(Периоды.Период, МЕСЯЦ) >= Фильтр.ДатаОкончания
		|			ТОГДА Фильтр.ДатаОкончания
		|		ИНАЧЕ КОНЕЦПЕРИОДА(Периоды.Период, МЕСЯЦ)
		|	КОНЕЦ КАК ДатаОкончания,
		|	Периоды.Период КАК Месяц,
		|	Фильтр.ДатаАктуальности КАК ДатаАктуальности
		|ПОМЕСТИТЬ ВТФильтОбработанныйИсполняемоеПредсталениеПлановоеВремя
		|ИЗ
		|	#ВТФильтр КАК Фильтр
		|		ЛЕВОЕ СОЕДИНЕНИЕ ВТПериодыДляВТФильтрИсполняемоеПредставлениеПлановоеВремя КАК Периоды
		|		ПО Периоды.Период МЕЖДУ Фильтр.ДатаНачала И Фильтр.ДатаОкончания";
	Иначе
		ТекстЗапросаВТПериоды = 
		"ВЫБРАТЬ
		|	Фильтр.Сотрудник КАК Сотрудник,
		|	Фильтр.ДатаНачала КАК ДатаНачала,
		|	Фильтр.ДатаОкончания КАК ДатаОкончания,
		|	НАЧАЛОПЕРИОДА(Фильтр.ДатаНачала, МЕСЯЦ) КАК Месяц,
		|	Фильтр.ДатаАктуальности КАК ДатаАктуальности
		|ПОМЕСТИТЬ ВТФильтОбработанныйИсполняемоеПредсталениеПлановоеВремя
		|ИЗ
		|	#ВТФильтр КАК Фильтр";
	КонецЕсли;
	ТекстЗапросаВТПериоды = СтрЗаменить(ТекстЗапросаВТПериоды, "#ВТФильтр", "ВТСотрудники");
	
	ЗапросВспомогательный.Текст = ТекстЗапросаВТПериоды;
	ЗапросВспомогательный.Выполнить();
	ПараметрыПолученияДанных.ИмяВТСотрудники = "ВТФильтОбработанныйИсполняемоеПредсталениеПлановоеВремя";
	
	ПараметрыПолученияДанных.ОтноситьПереходящуюЧастьСменыКДнюНачала = Истина;
	ПараметрыПолученияДанных.ПолучатьПлановоеВремяЗаПолныйМесяц = Истина;

	УчетРабочегоВремениРасширенный.СоздатьВТПлановоеВремя(
		Запрос.МенеджерВременныхТаблиц,
		Ложь,
		ПараметрыПолученияДанных);

	Запрос.Текст = 
	"ВЫБРАТЬ
	|	ПлановоеВремяСотрудников.Сотрудник КАК Сотрудник,
	|	ПлановоеВремяСотрудников.Дата КАК Дата,
	|	ПлановоеВремяСотрудников.ВидУчетаВремени КАК ВидУчетаВремени,
	|	ПлановоеВремяСотрудников.ДниПлан КАК Дни,
	|	ПлановоеВремяСотрудников.ЧасыПлан КАК Часы,
	|	ПлановоеВремяСотрудников.ЧасыНорма КАК ЧасыНорма,
	|	ПлановоеВремяСотрудников.Смена КАК Смена
	|ИЗ
	|	ВТПлановоеВремя КАК ПлановоеВремяСотрудников
	|
	|УПОРЯДОЧИТЬ ПО
	|	Сотрудник,
	|	Дата";
	Возврат Запрос.Выполнить();
КонецФункции
